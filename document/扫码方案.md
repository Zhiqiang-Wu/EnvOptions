```javascript
import {BrowserMultiFormatReader} from '@zxing/library';

const reader = new BrowserMultiFormatReader();
let deviceId;
reader.listVideoInputDevices().then((videoInputDevice) => {
    deviceId = videoInputDevice[0].deviceId;
    console.log('deviceId:'+ deviceId);
});
reader.decodeOnceFromVideoDevice(deviceId).then((re) => {
    console.log(re);
});
```
